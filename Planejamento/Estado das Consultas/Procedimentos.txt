-- Contar número total de clientes registrados.

CREATE OR REPLACE FUNCTION contar_clientes(OUT total integer)
  RETURNS integer AS
$BODY$
BEGIN
	total = COUNT(cod_cliente) FROM cliente;
END;
$BODY$
  LANGUAGE plpgsql
